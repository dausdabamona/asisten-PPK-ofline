```python
TIMELINE_ORDER = [
    ('SPESIFIKASI', 'tanggal_spesifikasi'),
    ('SURVEY', 'tanggal_survey_selesai'),
    ('HPS', 'tanggal_hps'),
    ('NOTA_DINAS_PP', 'tanggal_nota_dinas'),
    ('UNDANGAN', 'tanggal_undangan'),
    ('EVALUASI', 'tanggal_evaluasi'),
    ('NEGOSIASI', 'tanggal_negosiasi'),

    ('PENETAPAN_PENYEDIA', 'tanggal_penetapan'),
    ('SPK', 'tanggal_kontrak'),
    ('SPMK', 'tanggal_spmk'),
    ('PELAKSANAAN_MULAI', 'tanggal_mulai_kerja'),
    ('ADENDUM', 'tanggal_adendum'),  # Optional
    ('PELAKSANAAN_SELESAI', 'tanggal_selesai_kerja'),
    ('PEMERIKSAAN', 'tanggal_pemeriksaan'),
    ('BAHP_HASIL', 'tanggal_bahp_hasil'),
    ('BAST', 'tanggal_bast'),
    ('SPP', 'tanggal_spp'),
    ('SPM', 'tanggal_spm'),
    ('SP2D', 'tanggal_sp2d'),
]

def validate_timeline(paket_id, new_doc_type, new_date):
    """Validate new document date against existing timeline"""
    current_idx = get_timeline_index(new_doc_type)
    
    # Check all previous documents must have earlier dates
    for i, (doc_type, date_field) in enumerate(TIMELINE_ORDER[:current_idx]):
        existing_date = get_document_date(paket_id, doc_type)
        if existing_date and new_date < existing_date:
            raise ValidationError(
                f"Tanggal {new_doc_type} ({new_date}) tidak boleh sebelum "
                f"tanggal {doc_type} ({existing_date})"
            )
    
    # Check all subsequent documents must have later dates
    for i, (doc_type, date_field) in enumerate(TIMELINE_ORDER[current_idx+1:]):
        existing_date = get_document_date(paket_id, doc_type)
        if existing_date and new_date > existing_date:
            raise ValidationError(
                f"Tanggal {new_doc_type} ({new_date}) tidak boleh setelah "
                f"tanggal {doc_type} ({existing_date})"
            )
```
